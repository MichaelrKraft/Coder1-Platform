        // Show apply options modal
        function showApplyOptions() {
            const framework = detectFramework();
            const code = generateComponentCode(selectedComponent, framework);
            
            const modal = document.createElement('div');
            modal.className = 'fixed inset-0 bg-black/50 flex items-center justify-center z-50';
            modal.innerHTML = `
                

                    
Apply Component to Your Code

                    
                    

                        
Selected Element:

                        
${selectedElement.path}

                    

                    
                    

                        
Framework Detected:

                        
${framework.toUpperCase()}

                    

                    
                    

                        
                        
                        
                        
                        
                    

                    
                    

                        
Generated Code:

                        

                            
${code}

                        

                    

                    
                    
                

            `;
            document.body.appendChild(modal);
            
            // Store for functions
            window.generatedCode = code;
            window.selectedElementPath = selectedElement.path;
            window.componentName = selectedComponent.name;
        }

        // Copy terminal command
        window.copyTerminalCommand = function() {
            const command = `/replace "${window.selectedElementPath}" "${window.generatedCode.replace(/"/g, '\\"').replace(/\\n/g, '\\\\n')}"`;
            navigator.clipboard.writeText(command);
            alert('Terminal command copied! Paste it in the IDE terminal to apply the change.');
        };

        // Copy code with instructions
        window.copyCodeWithInstructions = function() {
            const instructions = `/* React Bits Component: ${window.componentName}
 * 
 * To integrate this component:
 * 1. Find the element: ${window.selectedElementPath}
 * 2. Replace it with the code below
 * 3. Make sure to import any required dependencies
 * 4. Adjust props and styling as needed
 */

${window.generatedCode}`;
            
            navigator.clipboard.writeText(instructions);
            alert('Code and instructions copied to clipboard!');
        };

        // Search functionality
        document.getElementById('searchInput').oninput = (e) => {
            const query = e.target.value.toLowerCase();
            document.querySelectorAll('.component-item').forEach(item => {
                const text = item.textContent.toLowerCase();
                item.style.display = text.includes(query) ? 'block' : 'none';
            });
        };

        // Category filters
        document.querySelectorAll('.category-btn').forEach(btn => {
            btn.onclick = () => {
                document.querySelectorAll('.category-btn').forEach(b => {
                    b.classList.remove('bg-purple-600/20', 'border-purple-600/30');
                    b.classList.add('bg-gray-600/20', 'border-gray-600/30');
                });
                btn.classList.remove('bg-gray-600/20', 'border-gray-600/30');
                btn.classList.add('bg-purple-600/20', 'border-purple-600/30');
                renderComponents(btn.dataset.category);
            };
        });

        // Refresh button
        document.getElementById('refreshBtn').onclick = () => {
            const frame = document.getElementById('previewFrame');
            if (frame.src) {
                frame.src = frame.src;
            } else {
                loadPreview();
            }
        };

        // Initialize
        renderComponents();
        
        // Add a timeout fallback in case detection hangs
        setTimeout(() => {
            const frame = document.getElementById('previewFrame');
            const loading = document.getElementById('loadingPreview');
            
            // If still loading after 3 seconds, force demo mode
            if (!frame.src && loading && !loading.classList.contains('hidden')) {
                console.log('Timeout reached, forcing demo mode');
                const statusEl = document.getElementById('previewStatus');
                statusEl.textContent = 'Demo Mode - Using sample website';
                statusEl.previousElementSibling.style.background = '#f59e0b';
                loadDemoSite();
            }
        }, 3000);
        
        loadPreview();
